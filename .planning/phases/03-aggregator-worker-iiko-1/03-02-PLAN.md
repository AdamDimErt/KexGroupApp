```typescript
import { NestFactory } from '@nestjs/core';
import { ValidationPipe } from '@nestjs/common';
import * as Sentry from '@sentry/node';
import { AppModule } from './app.module';

async function bootstrap() {
  // Sentry must init before NestFactory to capture early errors
  Sentry.init({
    dsn: process.env.SENTRY_DSN,
    environment: process.env.NODE_ENV || 'development',
    enabled: !!process.env.SENTRY_DSN,
  });

  const app = await NestFactory.create(AppModule);

  app.useGlobalPipes(
    new ValidationPipe({
      whitelist: true,
      forbidNonWhitelisted: true,
      transform: true,
    }),
  );

  const port = process.env.PORT ?? 3003;
  await app.listen(port, '0.0.0.0');
  console.log(`Aggregator Worker running on port ${port}`);
}
void bootstrap();
```

Key points:
- `import * as Sentry from '@sentry/node'` at the top
- `Sentry.init()` is the FIRST call inside bootstrap(), before NestFactory.create()
- `enabled: !!process.env.SENTRY_DSN` — Sentry is no-op when DSN is absent (dev/test environments)
- DSN from process.env.SENTRY_DSN (NEVER hardcoded)
- No performance monitoring (errors only, per CONTEXT.md decision)
